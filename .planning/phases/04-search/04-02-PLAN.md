```sql

```

Digests table schema:
```sql
CREATE TABLE IF NOT EXISTS digests (
  id TEXT PRIMARY KEY,
  period TEXT NOT NULL,

  stats TEXT NOT NULL,
  generated_at TEXT NOT NULL
```

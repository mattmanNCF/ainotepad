import { sqliteTable, text } from 'drizzle-orm/sqlite-core'

export const notes = sqliteTable('notes', {
  id: text('id').primaryKey(),
  rawText: text('raw_text').notNull(),
  submittedAt: text('submitted_at').notNull(),
  aiState: text('ai_state').notNull().default('pending'),
  aiAnnotation: text('ai_annotation'),
})

export type Note = typeof notes.$inferSelect
export type NewNote = typeof notes.$inferInsert

Create `src/main/db.ts`:

```typescript
import Database from 'better-sqlite3'
import { drizzle } from 'drizzle-orm/better-sqlite3'
import { app } from 'electron'
import path from 'path'
import * as schema from '../../drizzle/schema'

let _db: ReturnType<typeof drizzle> | null = null

export function getDb() {
  if (_db) return _db

  const dbPath = path.join(app.getPath('userData'), 'ainotepad.db')
  const sqlite = new Database(dbPath)

  // Enable WAL mode for better concurrent read performance
  sqlite.pragma('journal_mode = WAL')
  sqlite.pragma('foreign_keys = ON')

  // Create tables if they don't exist (simple inline migration for v1)
  sqlite.exec(`
    CREATE TABLE IF NOT EXISTS notes (
      id TEXT PRIMARY KEY,
      raw_text TEXT NOT NULL,
      submitted_at TEXT NOT NULL,
      ai_state TEXT NOT NULL DEFAULT 'pending',
      ai_annotation TEXT
    )
  `)

  _db = drizzle(sqlite, { schema })
  return _db
}
```

Create `src/main/ipc.ts`:

```typescript
import { ipcMain } from 'electron'
import { desc } from 'drizzle-orm'
import { v4 as uuidv4 } from 'uuid'
import { getDb } from './db'
import { notes } from '../../drizzle/schema'
import type { NoteRecord } from '../preload/index.d'

function toNoteRecord(row: typeof notes.$inferSelect): NoteRecord {
  return {
    id: row.id,
    rawText: row.rawText,
    submittedAt: row.submittedAt,
    aiState: row.aiState as NoteRecord['aiState'],
    aiAnnotation: row.aiAnnotation,
  }
}

export function registerIpcHandlers() {
  ipcMain.handle('notes:getAll', async () => {
    const db = getDb()
    const rows = await db.select().from(notes).orderBy(desc(notes.submittedAt))
    return rows.map(toNoteRecord)
  })

  ipcMain.handle('notes:create', async (_event, rawText: string) => {
    const db = getDb()
    const now = new Date().toISOString()
    const id = uuidv4()
    await db.insert(notes).values({ id, rawText, submittedAt: now })
    const [row] = await db.select().from(notes).where(notes.id.eq ? undefined : undefined)
    // Simpler: return constructed record directly
    const record: NoteRecord = {
      id,
      rawText,
      submittedAt: now,
      aiState: 'pending',
      aiAnnotation: null,
    }
    return record
  })
}
```
